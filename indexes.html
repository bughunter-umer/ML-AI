np

NumPy Cheat Sheet

Short, practical reference for array creation, linear algebra, randoms, statistics, reshaping and indexing — ready for ML practice.

Array creation

np.array([1,2,3])        # from list
np.zeros((2,3))          # zeros
np.ones((2,3))           # ones
np.arange(0,10,2)        # like range
np.linspace(0,1,5)       # n points evenly spaced

Random numbers

np.random.rand(2,3)      # uniform [0,1)
np.random.randn(3)       # normal (0,1)
np.random.randint(1,10,(2,2))  # ints [1,10)

Linear algebra

np.dot(a,b)             # dot product
np.matmul(A,B)           # matrix multiply (use @ too)
np.linalg.inv(A)         # inverse (square only)
np.linalg.pinv(A)        # pseudo-inverse
A.T or np.transpose(A)   # transpose

Statistics

np.mean(x)              # average
np.std(x)               # standard deviation
np.var(x)               # variance
np.min(x), np.max(x)    # min & max

Reshaping & dimension helpers

a.reshape(2,3)         # change shape
a.flatten()              # 1D copy
np.expand_dims(a,0)      # add new axis (batch)
a.squeeze()              # remove size-1 axes

Indexing & slicing

a[1:4]                 # slice
a[::2]                  # every 2nd
a[a>3]                  # boolean indexing
np.where(a>25)       # indices or ternary: np.where(cond, x,y)

Tips for ML

# normalize features
(x - x.mean(axis=0)) / x.std(axis=0)

# add bias column
X = np.concatenate([np.ones((n,1)), X], axis=1)

# linear regression closed form
theta = np.linalg.pinv(X.T @ X) @ X.T @ y
